const http = require('http');

const server = http.createServer((req, res) => {
  
  res.writeHead(200, {'Content-Type': 'text/html'});

  res.write(`
    
    
      College Website
    
    

      
ABC College

      
Welcome to our college website


      

      

    
    
  `);

  res.end();
});

server.listen(3000, () => {
  console.log("Server running at http://localhost:3000");
});
