Q16 - writeFileSync()


const fs = require('fs');

// Create student data
const students = [
  { name: "Devika", age: 21, course: "MCA" },
  { name: "Sam", age: 22, course: "BCA" }
];

// Convert to JSON string
const data = JSON.stringify(students, null, 2);

// Write file synchronously
fs.writeFileSync('students.json', data);

// Verify file creation
console.log("students.json file created successfully!");
