Q12 - Calculator


->calculator.js (Module File)

// Functions for calculator

function add(a, b) {
  return a + b;
}

function sub(a, b) {
  return a - b;
}

function mul(a, b) {
  return a * b;
}

function div(a, b) {
  return a / b;
}

// Export functions
module.exports = { add, sub, mul, div };

->app.js (Main File)
// Import module
const calc = require('./calculator');

// Using functions
console.log("Addition:", calc.add(5, 3));
console.log("Subtraction:", calc.sub(5, 3));
console.log("Multiplication:", calc.mul(5, 3));
console.log("Division:", calc.div(5, 3));
